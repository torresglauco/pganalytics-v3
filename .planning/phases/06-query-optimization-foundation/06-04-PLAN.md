    package metrics

    import (
        "regexp"
        "time"

        "github.com/gin-gonic/gin"
        "github.com/prometheus/client_golang/prometheus"
    )

    var (
        // Match UUIDs and numeric IDs in paths
        uuidPattern = regexp.MustCompile(`[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}`)
        idPattern   = regexp.MustCompile(`/\d+`)
    )

    // RequestCounter counts HTTP requests
    var RequestCounter = promauto.NewCounterVec(
        prometheus.CounterOpts{
            Name: "pganalytics_http_requests_total",
            Help: "Total number of HTTP requests",
        },
        []string{"method", "path", "status"},
    )

    // PrometheusMiddleware returns a Gin middleware that records metrics
    func PrometheusMiddleware() gin.HandlerFunc {
        return func(c *gin.Context) {
            start := time.Now()

            // Process request
            c.Next()

            // Record metrics after request completes
            duration := time.Since(start)

            // Normalize path (replace IDs with placeholders)
            path := normalizePath(c.FullPath())
            if path == "" {
                path = normalizePath(c.Request.URL.Path)
            }

            // Record to Prometheus
            status := c.Writer.Status()
            RecordAPIResponseTime(c.Request.Method, path, status, duration)

            // Increment request counter
            RequestCounter.WithLabelValues(
                c.Request.Method,
                path,
                strconv.Itoa(status),
            ).Inc()
        }
    }

    // normalizePath replaces UUIDs and numeric IDs with placeholders
    func normalizePath(path string) string {
        // Replace UUIDs
        path = uuidPattern.ReplaceAllString(path, ":uuid")
        // Replace numeric IDs
        path = idPattern.ReplaceAllString(path, "/:id")
        return path
    }
    ```

    package api

    import (
        "net/http"
        "time"

        "github.com/gin-gonic/gin"
        "github.com/torresglauco/pganalytics-v3/backend/internal/metrics"
    )

    // handleGetQueryStats returns query performance statistics
    // GET /api/v1/metrics/query-stats
    func (s *Server) handleGetQueryStats(c *gin.Context) {
        stats := metrics.GetGlobalQueryStats()

        c.JSON(http.StatusOK, gin.H{
            "count":            stats.Count,
            "min_duration":     stats.MinDuration.String(),
            "max_duration":     stats.MaxDuration.String(),
            "avg_duration":     stats.AvgDuration.String(),
            "p50":              stats.P50.String(),
            "p95":              stats.P95.String(),
            "p99":              stats.P99.String(),
            "min_duration_ms":  stats.MinDuration.Milliseconds(),
            "max_duration_ms":  stats.MaxDuration.Milliseconds(),
            "avg_duration_ms":  stats.AvgDuration.Milliseconds(),
            "p50_ms":           stats.P50.Milliseconds(),
            "p95_ms":           stats.P95.Milliseconds(),
            "p99_ms":           stats.P99.Milliseconds(),
        })
    }

    // handleGetHistogramBuckets returns histogram bucket configuration
    // GET /api/v1/metrics/histogram-buckets
    func (s *Server) handleGetHistogramBuckets(c *gin.Context) {
        buckets := metrics.HistogramBuckets()
        labels := metrics.PercentileLabels()

        bucketInfo := make([]gin.H, len(buckets))
        for i, b := range buckets {
            bucketInfo[i] = gin.H{
                "seconds": b,
                "label":   labels[b],
            }
        }

        c.JSON(http.StatusOK, gin.H{
            "buckets":     bucketInfo,
            "description": "Histogram buckets capture latency from 1ms to 10s",
        })
    }

    // handleGetMetricsSummary returns a combined metrics summary
    // GET /api/v1/metrics/summary
    func (s *Server) handleGetMetricsSummary(c *gin.Context) {
        queryStats := metrics.GetGlobalQueryStats()
        poolMetrics := gin.H{}

        if s.postgres != nil {
            poolMetrics["postgres"] = s.postgres.GetAllPoolMetrics()
        }
        if s.timescale != nil {
            poolMetrics["timescale"] = s.timescale.GetPoolMetrics()
        }

        c.JSON(http.StatusOK, gin.H{
            "query_stats":  queryStats,
            "pool_metrics": poolMetrics,
            "timestamp":    time.Now().UTC().Format(time.RFC3339),
        })
    }
    ```

    # Performance Monitoring

    pgAnalytics provides comprehensive performance monitoring through pprof and Prometheus metrics.

    ## pprof Profiling Endpoints

    Available at `/debug/pprof/*` for on-demand profiling:

    | Endpoint | Description |
    |----------|-------------|
    | `/debug/pprof/` | Index of all profiles |
    | `/debug/pprof/heap` | Memory allocation profile |
    | `/debug/pprof/profile?seconds=30` | CPU profile (30 second capture) |
    | `/debug/pprof/goroutine` | Goroutine stack traces |
    | `/debug/pprof/block` | Blocking profile |
    | `/debug/pprof/mutex` | Mutex contention profile |

    ### Usage Examples

    ```bash
    # Capture 30-second CPU profile
    curl -o cpu.prof "http://localhost:8080/debug/pprof/profile?seconds=30"
    go tool pprof -http=:8081 cpu.prof

    # Capture heap profile
    curl -o heap.prof "http://localhost:8080/debug/pprof/heap"
    go tool pprof -http=:8081 heap.prof

    # View goroutines
    curl "http://localhost:8080/debug/pprof/goroutine?debug=1"
    ```

    ## Prometheus Metrics

    Available at `/metrics` endpoint. Key metrics:

    ### API Performance
    - `pganalytics_api_response_time_seconds` - Histogram of API response times
    - `pganalytics_http_requests_total` - Counter of HTTP requests by method/path/status

    ### Query Performance
    - `pganalytics_query_duration_seconds` - Histogram of database query durations
    - `pganalytics_queries_total` - Counter of database queries

    ### Connection Pool
    - `pganalytics_pool_open_connections` - Current open connections
    - `pganalytics_pool_idle_connections` - Current idle connections
    - `pganalytics_pool_in_use_connections` - Connections currently in use
    - `pganalytics_pool_max_connections` - Maximum allowed connections

    ## Custom Metrics API

    JSON endpoints for programmatic access:

    | Endpoint | Description |
    |----------|-------------|
    | `GET /api/v1/metrics/query-stats` | Query performance percentiles (P50, P95, P99) |
    | `GET /api/v1/metrics/summary` | Combined metrics summary |
    | `GET /api/v1/system/pool-metrics` | Connection pool statistics |

    ## Grafana Dashboard

    Import the provided Grafana dashboard (see `grafana/dashboards/pganalytics.json`) for visualization.
