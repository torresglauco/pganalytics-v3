```go
package integration

import (
	"net/http/httptest"
	"testing"
	"time"

	"github.com/gin-gonic/gin"
	"github.com/golang-jwt/jwt/v5"
)

// authToken represents a JWT token for testing
type authToken struct {
	token string
}

// authenticateAs creates a request with authentication for the specified role
// Usage: req := authenticateAs(t, router, "admin")
func authenticateAs(t *testing.T, role string) string {
	t.Helper()

	// Create a test JWT token for the specified role
	claims := jwt.MapClaims{
		"user_id":  1,
		"username": "testuser",
		"role":     role,
		"exp":      time.Now().Add(1 * time.Hour).Unix(),
	}

	token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
	tokenString, err := token.SignedString([]byte("test-secret-key"))
	if err != nil {
		t.Fatalf("Failed to create test token: %v", err)
	}

	return "Bearer " + tokenString
}

// makeAuthenticatedRequest creates an authenticated HTTP test request
func makeAuthenticatedRequest(t *testing.T, router *gin.Engine, method, path string, body []byte, role string) *httptest.ResponseRecorder {
	t.Helper()

	req := httptest.NewRequest(method, path, bytes.NewReader(body))
	req.Header.Set("Content-Type", "application/json")
	req.Header.Set("Authorization", authenticateAs(t, role))

	w := httptest.NewRecorder()
	router.ServeHTTP(w, req)

	return w
}

// permissionTestCase represents a permission test scenario
type permissionTestCase struct {
	name           string
	method         string
	path           string
	body           []byte
	role           string
	expectedStatus int
	description    string
}

// runPermissionTests executes a table of permission tests
func runPermissionTests(t *testing.T, router *gin.Engine, tests []permissionTestCase) {
	t.Helper()

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			w := makeAuthenticatedRequest(t, router, tt.method, tt.path, tt.body, tt.role)
			assert.Equal(t, tt.expectedStatus, w.Code, tt.description)
		})
	}
}
```
