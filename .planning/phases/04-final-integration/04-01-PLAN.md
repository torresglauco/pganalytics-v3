<task type="auto" tdd="true">
  <name>Task 2: Create ESLint flat configuration</name>
  <files>frontend/eslint.config.mjs</files>
  <read_first>
    - frontend/package.json

  </read_first>
  <action>
    Create eslint.config.mjs using flat config format (not legacy .eslintrc format). This works with ESLint 8.56.0.

    The configuration must:
    1. Use flat config format (eslint.config.mjs with export default [])
    2. Apply to TypeScript files (.ts, .tsx) in src/ directory
    3. Configure TypeScript parser with JSX support
    4. Include these strict rules:
       - @typescript-eslint/no-unused-vars: error with argsIgnorePattern "^_"
       - @typescript-eslint/no-explicit-any: warn
       - react-hooks/rules-of-hooks: error
       - react-hooks/exhaustive-deps: warn
       - no-unused-vars: off (use TypeScript version instead)
    5. Ignore directories: dist/, node_modules/, coverage/, e2e/
    6. Detect React version automatically

    Example structure (adapt to project needs):
    ```javascript
    // frontend/eslint.config.mjs
    import js from '@eslint/js'
    import typescriptParser from '@typescript-eslint/parser'
    import typescript from '@typescript-eslint/eslint-plugin'
    import react from 'eslint-plugin-react'
    import reactHooks from 'eslint-plugin-react-hooks'

    export default [
      js.configs.recommended,
      {
        files: ['**/*.{ts,tsx}'],
        languageOptions: {
          parser: typescriptParser,
          parserOptions: {
            ecmaVersion: 2020,
            sourceType: 'module',
            ecmaFeatures: { jsx: true },
          },
        },
        plugins: {
          '@typescript-eslint': typescript,
          'react': react,
          'react-hooks': reactHooks,
        },
        rules: {
          // TypeScript strict rules
          '@typescript-eslint/no-unused-vars': ['error', { argsIgnorePattern: '^_' }],
          '@typescript-eslint/no-explicit-any': 'warn',
          'no-unused-vars': 'off',

          // React rules
          'react-hooks/rules-of-hooks': 'error',
          'react-hooks/exhaustive-deps': 'warn',
          'react/react-in-jsx-scope': 'off',
        },
        settings: {
          react: { version: 'detect' },
        },
      },
      {
        ignores: ['dist/', 'node_modules/', 'coverage/', 'e2e/'],
      },
    ]
    ```

    CRITICAL: Use .mjs extension and export default array format (flat config), NOT .eslintrc.json object format.
  </action>
