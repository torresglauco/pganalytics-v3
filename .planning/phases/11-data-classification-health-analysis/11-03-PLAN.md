Pattern:
```go
func (s *Server) TenantContextMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        userID, exists := c.Get("user_id")
        if !exists {
            c.Next()
            return
        }

        tenant, err := s.store.GetTenantByUserID(c.Request.Context(), userID.(uuid.UUID))
        if err != nil {
            c.JSON(403, gin.H{"error": "No tenant associated with user"})
            c.Abort()
            return
        }

        // Set tenant context for RLS
        err = s.store.SetTenantSessionVariable(c.Request.Context(), tenant.ID)
        if err != nil {
            c.JSON(500, gin.H{"error": "Failed to set tenant context"})
            c.Abort()
            return
        }

        c.Set("tenant_id", tenant.ID)
        c.Next()
    }
}
```
