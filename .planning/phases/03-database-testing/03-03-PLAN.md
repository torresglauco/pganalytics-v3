        if testing.Short() {

        testDB := testutil.NewTestDB(t)

        db, err := sql.Open("postgres", testDB.ConnStr)
        require.NoError(t, err)
        defer db.Close()

        // Create time-series table
        _, err = db.ExecContext(ctx, `
            CREATE TABLE IF NOT EXISTS ts_test (
                time TIMESTAMPTZ NOT NULL,
                value FLOAT
            )
        `)
        require.NoError(t, err)

        // Insert timestamps in UTC
        baseTime := time.Date(2026, 1, 15, 10, 0, 0, 0, time.UTC)
        timestamps := []time.Time{
            baseTime.Add(1 * time.Hour),
            baseTime.Add(2 * time.Hour),
            baseTime.Add(3 * time.Hour),
        }

        for i, ts := range timestamps {
            _, err = db.ExecContext(ctx, "INSERT INTO ts_test (time, value) VALUES ($1, $2)", ts, float64(i))

        for rows.Next() {
            var ts time.Time
            err := rows.Scan(&ts)
            require.NoError(t, err)
            assert.True(t, ts.After(prevTime), "Timestamps should be ordered")
