       func NewPostgresDB(connString string) (*PostgresDB, error) {
           ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
           defer cancel()

           // Parse config
           config, err := pgxpool.ParseConfig(connString)
           if err != nil {
               return nil, apperrors.DatabaseError("parse config", err.Error())
           }

           // Apply pool settings from environment
           maxConns := int32(100)
           if v := os.Getenv("MAX_DATABASE_CONNS"); v != "" {
               if m, err := strconv.ParseInt(v, 10, 32); err == nil && m > 0 {
                   maxConns = int32(m)
               }
           }
           config.MaxConns = maxConns

           minConns := int32(20)
           if v := os.Getenv("MAX_IDLE_DATABASE_CONNS"); v != "" {
               if m, err := strconv.ParseInt(v, 10, 32); err == nil && m > 0 {
                   minConns = int32(m)
               }
           }
           config.MinConns = minConns

           config.MaxConnLifetime = 15 * time.Minute
           config.MaxConnIdleTime = 10 * time.Minute

           // Create pool
           pool, err := pgxpool.NewWithConfig(ctx, config)
           if err != nil {
               return nil, apperrors.DatabaseError("create pool", err.Error())
           }

           // Test connection
           if err := pool.Ping(ctx); err != nil {
               return nil, apperrors.DatabaseError("ping database", err.Error())
           }

           // Create sql.DB wrapper for compatibility
           db := stdlib.OpenDBFromPool(pool)

           // Run migrations
           if err := runMigrations(ctx, db); err != nil {
               fmt.Fprintf(os.Stderr, "Warning: Migration error: %v\n", err)
           }

           return &PostgresDB{pool: pool, db: db}, nil
       }
       ```

       package metrics

       import (
           "github.com/prometheus/client_golang/prometheus"
           "github.com/prometheus/client_golang/prometheus/promauto"
           "github.com/torresglauco/pganalytics-v3/backend/internal/storage"
       )

       var (
           poolOpenConns = promauto.NewGaugeVec(prometheus.GaugeOpts{
               Name: "pganalytics_pool_open_connections",
               Help: "Number of open connections in the pool",
           }, []string{"database"})

           poolIdleConns = promauto.NewGaugeVec(prometheus.GaugeOpts{
               Name: "pganalytics_pool_idle_connections",
               Help: "Number of idle connections in the pool",
           }, []string{"database"})

           poolInUseConns = promauto.NewGaugeVec(prometheus.GaugeOpts{
               Name: "pganalytics_pool_in_use_connections",
               Help: "Number of connections in use",
           }, []string{"database"})

           poolMaxConns = promauto.NewGaugeVec(prometheus.GaugeOpts{
               Name: "pganalytics_pool_max_connections",
               Help: "Maximum connections allowed in the pool",
           }, []string{"database"})
       )

       // UpdatePoolMetrics updates Prometheus gauges from pool metrics
       func UpdatePoolMetrics(database string, metrics storage.PoolMetrics) {
           poolOpenConns.WithLabelValues(database).Set(float64(metrics.OpenConns))
           poolIdleConns.WithLabelValues(database).Set(float64(metrics.IdleConns))
           poolInUseConns.WithLabelValues(database).Set(float64(metrics.InUseConns))
           poolMaxConns.WithLabelValues(database).Set(float64(metrics.MaxOpenConns))
       }
       ```
